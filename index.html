Determine the optimal cutpoint of MMSE scores using some built in packages; gold standard: clinical diagnosis

import numpy as np
from sklearn.metrics import roc_curve, auc

# Assuming 'cutpointAnalysisMMSE' is your DataFrame with the necessary columns.
# Ensure the clinical diagnosis is binary coded
cutpointAnalysisMMSE['Clinical_Diagnosis_Binary'] = cutpointAnalysisMMSE['Clinical_Diagnosis-Simplified'].map({
    'Unimpaired': 0,
    'MCI': 1
})

# Invert MMSE scores so that lower scores become higher numbers and vice versa
cutpointAnalysisMMSE['Inverted_MMSE'] = 30 - cutpointAnalysisMMSE['NACCMMSE']

# Define true labels and scores (higher scores now indicate more impairment)
true_labels = cutpointAnalysisMMSE['Clinical_Diagnosis_Binary']
scores = cutpointAnalysisMMSE['Inverted_MMSE']

# Calculate the ROC curve
fpr, tpr, thresholds = roc_curve(true_labels, scores, pos_label=1)

# Calculate the AUC (Area Under the Curve)
roc_auc = auc(fpr, tpr)

# Find the optimal cutoff (where tpr is high and fpr is low, i.e., tpr - fpr is maximized)
optimal_idx = np.argmax(tpr - fpr)
optimal_threshold = 30 - thresholds[optimal_idx]  # Convert back to original MMSE scale

print("Area Under the ROC Curve (AUC):", roc_auc)
print("Optimal MMSE cutoff:", optimal_threshold)

def calculate_percentages(column):
    total_count = column.size
    value_counts = column.value_counts(dropna=False)
    percentages = (value_counts / total_count) * 100
    return percentages
